/**
 * Authentication Middleware
 */
export function jwtAuthenticate(req, res, next) {
	passport.authenticate(['jwt'], {
		session: false
	}, (err, user, info) => {
		if (err) {
			return next(err);
		}
		req.user = user || null;
		next();
	})(req, res, next);
}
